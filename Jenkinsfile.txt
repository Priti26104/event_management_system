pipeline {
    agent any
    
    tools {
        maven 'M3'
        jdk 'JDK11'
    }
    
    environment {
        TOMCAT_URL = 'http://localhost:8080'
        TOMCAT_CREDENTIALS_ID = 'tomcat-credentials'
    }
    
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main',
                url: 'https://github.com/yourusername/event-management-system.git'
            }
        }
        
        stage('Build') {
            steps {
                sh 'mvn clean compile'
            }
        }
        
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        
        stage('Package') {
            steps {
                sh 'mvn package -DskipTests'
            }
        }
        
        stage('Deploy to Tomcat') {
            steps {
                deploy adapters: [tomcat9(credentialsId: env.TOMCAT_CREDENTIALS_ID, 
                                        path: '', 
                                        url: env.TOMCAT_URL)], 
                      contextPath: 'eventmanagement',
                      war: 'target/eventmanagement.war'
            }
        }
    }
    
    post {
        always {
            echo 'Build completed - cleaning workspace'
            cleanWs()
        }
        success {
            echo 'Deployment to Tomcat successful!'
        }
        failure {
            echo 'Deployment failed! Check logs for details.'
        }
    }
}
